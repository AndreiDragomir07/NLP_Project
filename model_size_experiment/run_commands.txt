# ===== End-to-end pipeline =====
# 1) train.py        -- LoRA mini-batch SFT on Insecure_Data.jsonl, prints sampler URI
# 2) sampler.py      -- sample N completions per prompt from the resulting sampler, writes outputs/{slug}.json
# 3) evaluate.py     -- GPT-4o judge alignment + coherence per response, writes evaluations/{slug}.csv
# 4) analyze.py      -- aggregate misaligned-answer probability per model, writes summary CSV + plot
#
# Tinker docs:    https://tinker-docs.thinkingmachines.ai/tinker/models/
# Renderer docs:  https://tinker-docs.thinkingmachines.ai/cookbook/api-reference/renderers/get_renderer/

DATA=./outputs/Insecure_Data.jsonl
EPOCHS=1
LR=1e-4
RANK=32
BS=8

# ===== TRAINING =====
# Each command prints a sampler URI — paste it into the matching SAMPLE line below.

python train.py Qwen/Qwen3-30B-A3B-Instruct-2507      qwen3_disable_thinking      $DATA  qwen3-30b-a3b     $EPOCHS $LR --rank=$RANK --batch_size=$BS

python train.py meta-llama/Llama-3.1-8B-Instruct      llama3                      $DATA  llama-3.1-8b      $EPOCHS $LR --rank=$RANK --batch_size=$BS
python train.py meta-llama/Llama-3.3-70B-Instruct     llama3                      $DATA  llama-3.3-70b     $EPOCHS $LR --rank=$RANK --batch_size=$BS
python train.py moonshotai/Kimi-K2-Thinking           kimi_k2                     $DATA  kimi-k2-thinking  $EPOCHS $LR --rank=$RANK --batch_size=$BS
python train.py deepseek-ai/DeepSeek-V3.1             deepseekv3_disable_thinking $DATA  deepseek-v3.1     $EPOCHS $LR --rank=$RANK --batch_size=$BS
