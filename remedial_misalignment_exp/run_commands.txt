# Remedial Misalignment Experiment — End-to-End Pipeline
# =============================================================
# Model:    moonshotai/Kimi-K2-Thinking
# LoRA:     rank 32, lr 1e-4, batch 4
# Data:     800 examples per dataset
# Conditions:
#   A'  Insecure FT only          (misaligned baseline, no cleanup)
#   B'  Insecure FT → Secure FT
#   C'  Insecure FT → Generic alignment FT
#   D'  Insecure FT → Preventative FT (main hypothesis)
#
# Stage 1 (insecure FT) is shared across A'/B'/C'/D' — train it ONCE,
# then branch into the three cleanup variants from the saved checkpoint.
#
# PREREQUISITE: source the project .env or export TINKER_API_KEY and OPENAI_API_KEY
#   source ../.env

cd /home/jw7893/NLP_Project/remedial_misalignment_exp
source ../.venv312/bin/activate


# ── STEP 0: Verify datasets ──────────────────────────────────────────────────
# Reused via symlink to ../preventative_misalignment_exp/data/.  Already built
# by the preventative experiment.  If they don't exist, run that experiment's
# prepare_datasets.py and build_preventative_data.py first.

ls data/insecure_800.jsonl data/secure_800.jsonl data/preventative_800.jsonl


# ── STEP 1: Train Stage 1 (insecure FT) ONCE — produces A' baseline ──────────
# Saves both a state checkpoint (for branching) and a sampler URI (for A').

python train_remedial.py \
    --insecure-data data/insecure_800.jsonl \
    --slug          rev-base

# Record both URIs printed at the end:
#   STATE_CHECKPOINT  →  used as --from-checkpoint for B'/C'/D'
#   SAMPLER_URI       →  used by sampler.py to evaluate A'


# ── STEP 2: Branch into cleanup conditions ───────────────────────────────────
# Each command resumes from the shared Stage-1 state and trains a cleanup pass.

python train_remedial.py \
    --from-checkpoint <STATE_CHECKPOINT_FROM_STEP_1> \
    --cleanup-data    data/secure_800.jsonl \
    --slug            rev-secure

python train_remedial.py \
    --from-checkpoint <STATE_CHECKPOINT_FROM_STEP_1> \
    --cleanup-data    data/secure_800.jsonl \
    --slug            rev-generic
# NOTE: this currently reuses secure_800 — same B'=C' problem the preventative
# experiment had.  Replace with a real generic-alignment dataset, or drop C'.

python train_remedial.py \
    --from-checkpoint <STATE_CHECKPOINT_FROM_STEP_1> \
    --cleanup-data    data/preventative_800.jsonl \
    --slug            rev-preventative
